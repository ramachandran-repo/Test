async function waitForEvent(eventName) { return new Promise((resolve, reject) => { const eventHandler = (event) => { window.removeEventListener(eventName, eventHandler); resolve(event); }; window.addEventListener(eventName, eventHandler); }); } // Usage (async () => { try { console.log("Waiting for the event..."); const event = await waitForEvent('eventName'); // Replace 'eventName' with the actual event name console.log("Event occurred:", event); } catch (error) { console.error("Error occurred:", error); } })();
